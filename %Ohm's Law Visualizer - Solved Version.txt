%Ohm's Law Visualizer - Solved Version
% Demonstrates V = IR relationship for different resistances

% Define resistances (default values)
R_list = [1, 10, 100]; % Ohms

% Current range: 0 to 2 Amperes (200 points)
I = linspace(0, 2, 200); 

% Create figure
figure('Name', 'Ohm''s Law Demonstration', 'Color', 'white');
hold on;

% Colors for different resistance plots
colors = ['b', 'r', 'g', 'm', 'c']; 

% Plot V-I relationship for each resistance
for k = 1:length(R_list)
    V = I .* R_list(k); % V = IR
    plot(I, V, 'LineWidth', 2, 'Color', colors(k), 'DisplayName', sprintf('R = %d Ω', R_list(k)));
end

% Format plot
xlabel('Current (I) [A]', 'FontWeight', 'bold');
ylabel('Voltage (V) [V]', 'FontWeight', 'bold');
title('Ohm''s Law: Voltage vs. Current', 'FontSize', 12);
legend('Location', 'northwest');
grid on;
box on;
hold off;
